% --- Mathematical Formulas ---
\section*{Mathematical Formulas}

\subsection*{Core Identities}
$e^{i\pi} + 1 = 0$ \hfill $\sin^2\theta + \cos^2\theta = 1$ \\
$\frac{d}{dx}(x^n) = nx^{n-1}$ \hfill $\int x^n\,dx = \frac{x^{n+1}}{n+1}+C$ \\
$(a+b)^2 = a^2 + 2ab + b^2$ \hfill $a^2 - b^2 = (a+b)(a-b)$

\subsection*{Trigonometry}
$\sin(a \pm b) = \sin a \cos b \pm \cos a \sin b$ \\
$\cos(a \pm b) = \cos a \cos b \mp \sin a \sin b$ \\
$\tan(a \pm b) = \frac{\tan a \pm \tan b}{1 \mp \tan a \tan b}$ \hfill $\tan\theta = \frac{\sin\theta}{\cos\theta}$

\subsection*{Probability \& Statistics}
$P(A|B) = \frac{P(B|A)P(A)}{P(B)}$ \hfill Bayes' Theorem \\
$\mathbb{E}[X] = \sum_i x_i p(x_i)$ \hfill $\text{Var}(X) = \mathbb{E}[X^2] - (\mathbb{E}[X])^2$ \\
$\text{Cov}(X,Y) = \mathbb{E}[XY] - \mathbb{E}[X]\mathbb{E}[Y]$ \\
$\rho_{X,Y} = \frac{\text{Cov}(X,Y)}{\sigma_X \sigma_Y}$ \hfill Correlation

\begin{tipbox}
\textbf{Tip:} $\text{Var}(aX+b) = a^2\text{Var}(X)$ -- constant shift doesn't affect variance
\end{tipbox}

\subsection*{Calculus}
$\frac{d}{dx}\sin x = \cos x$ \hfill $\frac{d}{dx}\cos x = -\sin x$ \\
$\frac{d}{dx}e^x = e^x$ \hfill $\frac{d}{dx}\ln x = \frac{1}{x}$ \\
$\int e^x dx = e^x + C$ \hfill $\int \frac{1}{x} dx = \ln|x| + C$ \\
Chain rule: $\frac{d}{dx}f(g(x)) = f'(g(x))g'(x)$ \\
Product: $(uv)' = u'v + uv'$ \hfill Quotient: $(\tfrac{u}{v})' = \tfrac{u'v-uv'}{v^2}$

\subsection*{Linear Algebra}
$\mathbf{A}^{-1}\mathbf{A} = \mathbf{I}$ \hfill $(\mathbf{AB})^T = \mathbf{B}^T\mathbf{A}^T$ \\
$\det(\mathbf{AB}) = \det(\mathbf{A})\det(\mathbf{B})$ \\
Eigenvalue: $\mathbf{A}\mathbf{v} = \lambda\mathbf{v}$ \hfill $\det(\mathbf{A}-\lambda\mathbf{I}) = 0$ \\
SVD: $\mathbf{A} = \mathbf{U}\mathbf{\Sigma}\mathbf{V}^T$ \hfill Rank $\leq \min(m,n)$

\section*{Machine Learning Essentials}

\subsection*{Loss Functions}
\textbf{MSE:} $\frac{1}{n}\sum(y_i - \hat{y}_i)^2$ \hfill Regression \\
\textbf{MAE:} $\frac{1}{n}\sum|y_i - \hat{y}_i|$ \hfill Robust to outliers \\
\textbf{Cross-Entropy:} $-\sum y_i \log(\hat{y}_i)$ \hfill Classification \\
\textbf{Hinge:} $\max(0, 1-y_i\hat{y}_i)$ \hfill SVM

\subsection*{Activation Functions}
$\sigma(x) = \frac{1}{1+e^{-x}}$ \hfill Sigmoid (0 to 1) \\
$\tanh(x) = \frac{e^x-e^{-x}}{e^x+e^{-x}}$ \hfill Tanh (-1 to 1) \\
$\text{ReLU}(x) = \max(0, x)$ \hfill Most common \\
$\text{LeakyReLU}(x) = \max(0.01x, x)$ \hfill Prevents dead neurons \\
$\text{Softmax}(x_i) = \frac{e^{x_i}}{\sum_j e^{x_j}}$ \hfill Multi-class output

\subsection*{Optimization}
\textbf{SGD:} $\theta \leftarrow \theta - \eta \nabla L$ \\
\textbf{Momentum:} $v \leftarrow \beta v + \nabla L$, $\theta \leftarrow \theta - \eta v$ \\
\textbf{Adam:} Adaptive learning rates per parameter \\
\textbf{Learning rate decay:} Reduce $\eta$ over time

\subsection*{Regularization}
\textbf{L1:} $\lambda \sum |\theta_i|$ \hfill Sparse weights \\
\textbf{L2:} $\lambda \sum \theta_i^2$ \hfill Weight decay \\
\textbf{Dropout:} Randomly zero neurons during training \\
\textbf{Batch Norm:} Normalize layer inputs \\
\textbf{Data Aug:} Augment training data artificially

\subsection*{Evaluation Metrics}
\textbf{Accuracy:} $\frac{TP+TN}{TP+TN+FP+FN}$ \\
\textbf{Precision:} $\frac{TP}{TP+FP}$ \hfill Of predicted pos, how many correct \\
\textbf{Recall:} $\frac{TP}{TP+FN}$ \hfill Of actual pos, how many found \\
\textbf{F1:} $\frac{2 \cdot \text{Prec} \cdot \text{Rec}}{\text{Prec} + \text{Rec}}$ \hfill Harmonic mean \\
\textbf{AUC-ROC:} Area under ROC curve

\subsection*{Model Types}
\textbf{Linear Reg:} $\hat{y} = \mathbf{w}^T\mathbf{x} + b$ \\
\textbf{Logistic Reg:} $P(y=1) = \sigma(\mathbf{w}^T\mathbf{x})$ \\
\textbf{Decision Tree:} Recursive splits on features \\
\textbf{Random Forest:} Ensemble of trees, vote/average \\
\textbf{SVM:} Max-margin hyperplane separator \\
\textbf{k-NN:} Classify by k nearest neighbors \\
\textbf{Neural Net:} Layers of learned transformations
