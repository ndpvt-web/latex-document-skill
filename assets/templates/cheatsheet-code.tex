class Dog(Animal):
    def __init__(self, name):
        super().__init__()
        self.name = name

    def speak(self) -> str:
        return f"{self.name} barks"

    @property
    def info(self):
        return self._info

    @classmethod
    def create(cls, name):
        return cls(name)

    @staticmethod
    def species():
        return "Canis"

\begin{codeblock}{Dataclasses}

from dataclasses import (
    dataclass, field
)

@dataclass
class Point:
    x: float
    y: float
    label: str = ""
    tags: list = field(
        default_factory=list
    )

    def dist(self, other):
        return ((self.x-other.x)**2
              + (self.y-other.y)**2)**0.5
